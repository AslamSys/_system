### Configura√ß√£o NATS 1
```conf

  name: mordomo-cluster

    nats-route://nats-2:6222

  domain: "mordomo"

  users = [
    {
      user: "admin"
      password: "$2a$11$..." # bcrypt hash

        publish: [">"]
        subscribe: [">"]
      }
    },

      user: "mordomo"
      password: "$2a$11$..."
      permissions: {
        publish: ["audio.>", "brain.>", "tts.>", "discovery.>"]
        subscribe: ["audio.>", "brain.>", "tts.>", "discovery.>"]
      }
    }
  ]
}

max_subscriptions: 0

# Logging

trace: false
logtime: true
```
